---
title: 单选框组
description: 在一组相关且互斥数据中，用户仅能选择一个选项。
---

## 演示

```jsx demo online=RadioGroup
import { Radio, RadioGroup } from "@resolid/react-ui";

export default function App() {
  return (
    <RadioGroup name={"frameworksRadioGroup"} className={"flex flex-row items-center gap-5"}>
      <Radio value={"React"}>React</Radio>
      <Radio value={"Angular"}>Angular</Radio>
      <Radio value={"Vue"}>Vue</Radio>
      <Radio value={"SolidJS"}>SolidJS</Radio>
      <Radio value={"Svelte"}>Svelte</Radio>
    </RadioGroup>
  );
}
```

## 用法

```js
import { RadioGroup, Radio } from "@resolid/react-ui";
```

- `RadioGroup`: 单选框组
- `Radio`: 单选框，必须在 `RadioGroup` 内部使用

## 特点

- 遵循 [WAI ARIA Radio Group](https://www.w3.org/WAI/ARIA/apg/patterns/radiobutton/) 设计模式。
- 每个 `Radio` 都使用本机 HTML `<input>` 元素构建，该元素在视觉上隐藏以允许自定义样式。
- 完整的键盘导航。
- 支持水平/垂直方向。
- 可以是受控的，也可以是不受控的。

## 举例

### 可控

使用 `RadioGroup` 组件的 `value` 和 `onChange` 属性来控制选定的单选项目

```jsx demo
import { Radio, RadioGroup } from "@resolid/react-ui";
import { useState } from "react";

export default function App() {
  const [value, setValue] = useState("");

  return (
    <div className={"flex flex-col gap-2"}>
      <p>你选择的框架是: {value}</p>
      <RadioGroup
        name={"frameworksRadioGroup"}
        value={value}
        onChange={setValue}
        className={"flex flex-row items-center gap-5"}
      >
        <Radio value={"React"}>React</Radio>
        <Radio value={"Angular"}>Angular</Radio>
        <Radio value={"Vue"}>Vue</Radio>
        <Radio value={"SolidJS"}>SolidJS</Radio>
        <Radio value={"Svelte"}>Svelte</Radio>
      </RadioGroup>
    </div>
  );
}
```

### 大小

通过传递 `size` 属性来更改单选框的大小，单选框支持 `xs`、`sm`、`md`、`lg` 和 `xl` 这些大小。

```jsx demo
import { Radio, RadioGroup } from "@resolid/react-ui";

export default function App() {
  return (
    <RadioGroup
      name={"sizesRadioGroup"}
      defaultValue={"md"}
      className={"flex flex-row items-center gap-5"}
    >
      <Radio value={"xs"} size={"xs"}>
        XS
      </Radio>
      <Radio value={"sm"} size={"sm"}>
        SM
      </Radio>
      <Radio value={"md"}>MD</Radio>
      <Radio value={"lg"} size={"lg"}>
        LG
      </Radio>
      <Radio value={"xl"} size={"xl"}>
        XL
      </Radio>
    </RadioGroup>
  );
}
```

### 颜色

通过传递 `color` 属性来更改单选框的颜色，单选框支持 `primary`、`secondary`、`success`、`warning`、`danger` 和 `neutral` 这些颜色。

```jsx demo
import { Radio, RadioGroup } from "@resolid/react-ui";

export default function App() {
  const colors = ["primary", "secondary", "success", "warning", "danger", "neutral"];

  return (
    <RadioGroup
      name={"colorsRadioGroup"}
      defaultValue={"secondary"}
      className={"flex flex-row items-center gap-5"}
    >
      {colors.map((color) => (
        <Radio key={color} value={color} color={color}>
          {color}
        </Radio>
      ))}
    </RadioGroup>
  );
}
```
