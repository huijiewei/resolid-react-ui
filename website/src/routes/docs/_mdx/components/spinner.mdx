## 演示

```jsx demo online=Spinner componentProps=spinner/spinner.tsx|color,size
import { Spinner } from "@resolid/react-ui";

export default function App() {
  return <Spinner />;
}
```

## 用法

```js
import { Spinner } from "@resolid/react-ui";
```

## 举例

### 颜色

通过传递 `color` 属性来更改加载器的颜色。加载器支持 `primary`、`secondary`、`success`、`warning`、`danger` 和 `neutral` 这些颜色。

```jsx demo
import { Spinner } from "@resolid/react-ui";

export default function App() {
  const colors = ["primary", "secondary", "success", "warning", "danger", "neutral"];

  return (
    <div className={"flex flex-row gap-3"}>
      {colors.map((color) => (
        <Spinner key={color} color={color} />
      ))}
    </div>
  );
}
```

### 大小

通过传递 `size` 属性来更改加载器的大小。加载器支持 `xs`、`sm`、`md`、`lg` 和 `xl` 这些大小。

```jsx demo
import { Spinner } from "@resolid/react-ui";

export default function App() {
  return (
    <div className={"flex flex-row items-center gap-3"}>
      <Spinner size={"xs"} />
      <Spinner size={"sm"} />
      <Spinner size={"md"} />
      <Spinner size={"lg"} />
      <Spinner size={"xl"} />
    </div>
  );
}
```
