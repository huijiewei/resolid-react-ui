---
title: 开关
description: 互斥性的操作控件，用户可打开或关闭某个功能。
---

## 演示

```jsx demo online=Switch componentProps=switch/switch.tsx|color,size,disabled,readOnly,invalid
import { Switch } from "@resolid/react-ui";

export default function App() {
  return <Switch defaultChecked>开关</Switch>;
}
```

## 用法

```js
import { Switch } from "@resolid/react-ui";
```

## 特点

- 遵循 [WAI ARIA Switch](https://www.w3.org/WAI/ARIA/apg/patterns/switch/) 设计模式。
- 由本机 HTML `<input>` 元素构建，该元素在视觉上隐藏以允许自定义样式。
- 标记对辅助技术的支持。
- 可以是受控的，也可以是不受控的。

## 举例

### 可控

`value` 属性可用于可控值。当用户输入并接收新值时， `onChange` 事件将被触发。

```jsx demo
import { Switch } from "@resolid/react-ui";
import { useState } from "react";

export default function App() {
  const [checked, setChecked] = useState(false);

  return (
    <div className={"flex flex-col gap-2"}>
      <p>飞行模式: {checked ? <span className="text-fg-success">已激活</span> : "未激活"}.</p>
      <Switch checked={checked} onChange={setChecked}>
        飞行模式
      </Switch>
    </div>
  );
}
```

### 大小

通过传递 `size` 属性来更改开关的大小。开关支持 `xs`、`sm`、`md`、`lg` 和 `xl` 这些大小。

```jsx demo
import { Switch } from "@resolid/react-ui";

export default function App() {
  return (
    <div className={"flex flex-row items-center gap-3"}>
      <Switch size={"xs"}>XS</Switch>
      <Switch size={"sm"}>SM</Switch>
      <Switch>MD</Switch>
      <Switch size={"lg"}>LG</Switch>
      <Switch size={"xl"}>XL</Switch>
    </div>
  );
}
```

### 颜色

通过传递 `color` 属性来更改开关的颜色。开关支持 `primary`、`secondary`、`success`、`warning`、`danger` 和 `neutral` 这些颜色。

```jsx demo
import { Switch } from "@resolid/react-ui";

export default function App() {
  const colors = ["primary", "secondary", "success", "warning", "danger", "neutral"];

  return (
    <div className={"flex flex-row items-center gap-3"}>
      {colors.map((color) => (
        <Switch key={color} defaultChecked color={color} size={"sm"}>
          {color}
        </Switch>
      ))}
    </div>
  );
}
```
