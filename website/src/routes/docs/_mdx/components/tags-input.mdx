---
title: 标签输入
description: 允许用户向输入字段添加标签的组件。
---

## 演示

```jsx demo online=TagsInput componentProps=tags-input/tags-input.tsx|size,disabled,readOnly,invalid
import { TagsInput, TagsInputInput } from "@resolid/react-ui";

export default function App() {
  return (
    <TagsInput placeholder={"请输入标签"} className={"max-w-80"} defaultValue={["React", "Vue"]}>
      <TagsInputInput />
    </TagsInput>
  );
}
```

## 用法

```jsx
import { TagsInput, TagsInputInput } from "@resolid/react-ui";
```

- `TagsInput`: 标签输入的根容器。
- `TagsInputInput`: 标签输入的输入框。

## 特点

- 可以是受控的，也可以是不受控的。
- 完整的键盘导航。
- 限制标签的数量。
- 接受剪贴板中的值。

## 举例

### 可控值

使用 `value` 和 `onChange` 属性来控制输入的值。

```jsx demo
import { TagsInput, TagsInputInput } from "@resolid/react-ui";
import { useState } from "react";

export default function App() {
  const [value, setValue] = useState(["React", "Vue"]);

  return (
    <div className={"flex flex-col gap-2"}>
      <p>你输入了标签: {value.join(", ")}</p>
      <TagsInput
        name={"tagsTest"}
        placeholder={"请输入标签"}
        className={"max-w-80"}
        value={value}
        onChange={setValue}
      >
        <TagsInputInput />
      </TagsInput>
    </div>
  );
}
```

### 限制数量

使用 `max` 属性可以限制标签输入的数量

```jsx demo
import { TagsInput, TagsInputInput } from "@resolid/react-ui";

export default function App() {
  return (
    <TagsInput
      placeholder={"请输入标签"}
      max={3}
      className={"max-w-80"}
      defaultValue={["React", "Vue"]}
    >
      <TagsInputInput />
    </TagsInput>
  );
}
```

### 限制长度

`TagsInputInput` 的 `maxLength` 属性可以限制标签输入的长度

```jsx demo
import { TagsInput, TagsInputInput } from "@resolid/react-ui";

export default function App() {
  return (
    <TagsInput placeholder={"请输入标签"} className={"max-w-80"} defaultValue={["React", "Vue"]}>
      <TagsInputInput maxLength={6} />
    </TagsInput>
  );
}
```

### 失去焦点增加

使用 `addOnBlur` 属性可以允许输入框失去焦点时添加标签

```jsx demo
import { TagsInput, TagsInputInput } from "@resolid/react-ui";

export default function App() {
  return (
    <TagsInput
      placeholder={"请输入标签"}
      addOnBlur
      className={"max-w-80"}
      defaultValue={["React", "Vue"]}
    >
      <TagsInputInput />
    </TagsInput>
  );
}
```

### 允许粘贴

使用 `addOnPaste` 属性可以允许接受剪贴板中的值

```jsx demo
import { TagsInput, TagsInputInput } from "@resolid/react-ui";

export default function App() {
  return (
    <TagsInput
      placeholder={"请输入标签"}
      addOnPaste
      className={"max-w-80"}
      defaultValue={["React", "Vue"]}
    >
      <TagsInputInput />
    </TagsInput>
  );
}
```
